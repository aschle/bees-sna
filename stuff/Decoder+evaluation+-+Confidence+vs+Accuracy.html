def int_id_to_binary( id ):
    result = np.zeros( 12, dtype = np.int )
    a = 11
    while id:
        result[ a ] = id & 1
        id >>= 1
        a -= 1
    return result

def truth_generator(batch_size):
    idx = 0 
    while True:
        if idx + batch_size > truth_tags.shape[0]:
            idx = 0
        tags = (truth_tags[idx:idx+batch_size] / 255.)
        tags = gaussian_filter1d(tags, 2/3, axis=-1)
        tags = gaussian_filter1d(tags, 2/3, axis=-2)
        tags = tags * 2 - 1
        
        labels = list(np.array([int_id_to_binary(i)[::-1] for i in truth_labels[idx:idx+batch_size]]).T)
        
        yield tags[:, np.newaxis], labels + [np.zeros(batch_size) for _ in range(6)]
        idx += batch_size

def hist_equalisation(generator):
    for data, labels in generator:
        eq_imgs = []
        for img in data:
            eq_imgs.append(2*equalize_hist(img) - 1)
        yield np.array(eq_imgs), labels

def hamming_distance(y_true, y_pred):
    return (y_true != y_pred).sum(axis=-1)


def mean_hamming_distance(y_true, y_pred):
    nb_intermediate = np.sum(y_true == 0.5)
    print(nb_intermediate)
    hd = hamming_distance(y_true, y_pred)
    return (np.sum(hd) - nb_intermediate) / len(hd) # (len(hd) - nb_intermediate)


def nth_bit_right_accuracy(bits_true, bits_pred, nth):
    accurate = np.sum(bits_true == bits_pred, axis=1) >= nth
    return accurate.mean()


def mse(x, y):
    return ((x - y)**2).mean(axis=-1)
